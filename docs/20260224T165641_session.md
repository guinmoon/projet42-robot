> ## 3️⃣ Краткое примерное объявление одной новой анимации
> 
> ```cpp
> // src/eyes_anim_heart.cpp
> #include "eyes_drawer.h"
> #include "eyes_anim_base.h"
> 
> void EyesAnimator::anim_starburst(uint16_t color, uint16_t durationMs) noexcept
> {
>     const uint16_t half = durationMs / 2;
>     const uint16_t stepDelay = 30;
>     const uint16_t steps = half / stepDelay;
> 
>     for (uint16_t i = 0; i < steps; ++i) {
>         // Радиус растёт от 2 до 30 пикселей (пример)
>         uint16_t radius = map(i, 0, steps, 2, 30);
>         // Позиция – центр глаза (чтение из LuLuEyes)
>         int16_t cx = eyes->eyeCenterX();
>         int16_t cy = eyes->eyeCenterY();
> 
>         // рисуем несколько лучей‑звёздочек
>         for (uint8_t j = 0; j < 8; ++j) {
>             float angle = j * PI / 4.0f + (i * 0.01f);
>             int16_t x = cx + (int16_t)(radius * cosf(angle));
>             int16_t y = cy + (int16_t)(radius * sinf(angle));
>             drawPixel(x, y, color);
>         }
>         vTaskDelay(pdMS_TO_TICKS(stepDelay));
>     }
> 
>     // «затухание» – просто стираем, оставляя фон
>     eyes->clearEyes();
> }
> ```
> 
> *`drawPixel` реализуется в `eyes_anim_base.cpp` и использует уже подготовленный `LGFX`‑контекст (`eyes->gfx`).*
